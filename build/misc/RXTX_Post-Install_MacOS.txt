################################
# RXTX Post-Installation Steps
# execute using sudo 
################################

mkdir /var/lock
chgrp uucp /var/lock
chmod 775 /var/lock

# Mac OS 10.5.x 
dscl . -read /groups/_uucp users | grep <user>
dscl . -append /groups/_uucp GroupMembership <user>

# Mac OS X 10.6.x (Snow Leopard)
niutil -readprop / /groups/uucp users | grep <user>
niutil -mergeprop / /groups/uucp users <user>

chmod g+w /var/spool/uucp/
